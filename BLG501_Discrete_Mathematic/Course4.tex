\section{Ackermann function}

\[
    A(m, n) =
    \begin{cases}
        A(0, n) = n+1 \\
        A(m, 0) = A(m-1, 1) \\
        A(m, n) = A(m-1, A(m, n-1))
    \end{cases}
\]

\[
    A(1, n) =
    \begin{cases}
        A(0, A(1,n-1)) \\
        A(1, n-1) + 1 \\
        A(1, 0) + n \\
        A(0, 1) + n
    \end{cases}
    = n + 2
\]

\[
    A(1,n) = n+2 \\
    A(2,n) = 2n + 3 \\
    A(3,n) = 2^n - 3
\]
